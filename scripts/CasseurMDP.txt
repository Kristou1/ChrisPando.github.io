#! /bin/bash

fichier="/usr/share/utilisateurs.txt"

cat $fichier |while read prenom_date_; do
    echo "Essai de : $prenom"
    output=$(unzip -P $prenom /usr/share/secret.zip)
    echo $output

    echo "Essai de : $prenom$date"
    output=$(unzip -P $prenom$date /usr/share/secret.zip)
    echo $output
done
